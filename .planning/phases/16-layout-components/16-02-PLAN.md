Create the `src/components/MobileViewToggle/` directory and `MobileViewToggle.tsx`.

The component is a floating action button rendered on the map/preview pane. It must be:
- A controlled component accepting `activeView` and `onToggle` props
- Only visible on mobile tier (returns null on tablet/desktop)
- Styled as a floating button positioned in the top-right corner
- Has a minimum 44px touch target (TOUCH-01 preparedness)
- Uses the `useBreakpoint` hook for tier detection

```tsx
import { useBreakpoint } from '../../hooks/useBreakpoint';

interface MobileViewToggleProps {
  activeView: 'map' | 'preview';
  onToggle: () => void;
}

export function MobileViewToggle({ activeView, onToggle }: MobileViewToggleProps) {
  const tier = useBreakpoint();

  if (tier !== 'mobile') return null;

  return (
    <button
      onClick={onToggle}
      aria-label={activeView === 'map' ? 'Show 3D preview' : 'Show map'}
      data-testid="mobile-view-toggle"
      style={{
        position: 'absolute',
        top: 'max(12px, var(--safe-top, 12px))',
        right: 12,
        zIndex: 20,
        minHeight: 44,
        minWidth: 44,
        padding: '8px 16px',
        backgroundColor: 'rgba(17, 24, 39, 0.85)',
        backdropFilter: 'blur(8px)',
        border: '1px solid rgba(255,255,255,0.15)',
        borderRadius: '8px',
        color: '#e5e7eb',
        fontSize: '13px',
        fontWeight: 600,
        cursor: 'pointer',
        boxShadow: '0 2px 8px rgba(0,0,0,0.3)',
        touchAction: 'manipulation',
      }}
    >
      {activeView === 'map' ? '3D Preview' : 'Map'}
    </button>
  );
}
```

Key decisions:
- **Controlled component:** Accepts `activeView` and `onToggle` as props (not store-driven). SplitLayout manages `activeTab` state and will pass it through in Plan 03.
- **`tier !== 'mobile'` guard:** Returns null on tablet/desktop — the toggle only exists on mobile where map and preview are full-screen layers.
- **`minHeight: 44` + `minWidth: 44`:** Meets 44px touch target requirement.
- **`touchAction: 'manipulation'`:** Prevents browser zoom on double-tap, ensuring quick taps are responsive.
- **`position: absolute`:** Floats over the map/preview pane. Not inside the bottom sheet — it must be visible at all sheet heights.
- **`aria-label`:** Descriptive label for accessibility — states what will happen on tap.
- **Visual treatment:** Matches existing app aesthetic (dark translucent background, blur, subtle border).
- **Text label:** Shows what the button will switch TO: "3D Preview" when on map, "Map" when on preview.

import { render, screen, fireEvent } from '@testing-library/react';
import { describe, it, expect, vi } from 'vitest';

describe('MobileViewToggle', () => {
