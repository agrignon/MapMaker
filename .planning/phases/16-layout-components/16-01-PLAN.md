**Step 2: Create `src/components/BottomSheet/BottomSheet.tsx`**

Create the `src/components/BottomSheet/` directory and the component file.

The component:
- Import `{ Drawer }` from `'vaul'` and `{ useState, useCallback }` from `'react'`
- Define a constant `PEEK_PX = 80` for the peek snap point
- Define snap points as `[PEEK_PX, 0.45, 1]` — vaul treats numbers 0-1 as fractions of `window.innerHeight`. The value `0.45` gives approximately 45vh. The value `1` gives full height (vaul internally caps and handles safe areas). This avoids the problem of vaul not parsing CSS viewport unit strings.
- Accept `{ children }: { children: React.ReactNode }` props
- Use `useState` for `snap` initialized to `PEEK_PX` (sheet starts at peek)
- Handle the double-tap-closes-sheet pitfall (GitHub Issue #362): Add `onOpenChange` callback that prevents the sheet from closing by calling nothing (or re-setting snap to PEEK_PX if closed)

```tsx
import { useState, useCallback } from 'react';
import { Drawer } from 'vaul';

const PEEK_PX = 80;
const SNAP_POINTS = [PEEK_PX, 0.45, 1] as const;

export function BottomSheet({ children }: { children: React.ReactNode }) {
  const [snap, setSnap] = useState<number | string | null>(PEEK_PX);

  // Safety net: if vaul fires onOpenChange(false) due to double-tap on handle
  // (GitHub #362), prevent the sheet from disappearing by ignoring close.
  const handleOpenChange = useCallback((open: boolean) => {
    if (!open) {
      // Re-snap to peek instead of closing — there is no Trigger to reopen
      setSnap(PEEK_PX);
    }
  }, []);

  return (
    <Drawer.Root
      open={true}
      modal={false}
      dismissible={false}
      snapPoints={SNAP_POINTS as unknown as (number | string)[]}
      activeSnapPoint={snap}
      setActiveSnapPoint={setSnap}
      fadeFromIndex={2}
      onOpenChange={handleOpenChange}
    >
      <Drawer.Portal>
        <Drawer.Content
          data-testid="bottom-sheet-content"
          style={{
            position: 'fixed',
            bottom: 0,
            left: 0,
            right: 0,
            zIndex: 10,
            backgroundColor: 'rgba(17, 24, 39, 0.95)',
            backdropFilter: 'blur(12px)',
            borderRadius: '14px 14px 0 0',
            boxShadow: '0 -4px 20px rgba(0,0,0,0.4)',
            borderTop: '1px solid rgba(255,255,255,0.15)',
            paddingBottom: 'max(8px, var(--safe-bottom))',
            paddingLeft: 'var(--safe-left)',
            paddingRight: 'var(--safe-right)',
            display: 'flex',
            flexDirection: 'column',
          }}
        >
          <Drawer.Handle
            data-testid="bottom-sheet-handle"
            style={{
              width: 36,
              height: 4,
              borderRadius: 2,
              backgroundColor: 'rgba(255,255,255,0.3)',
              margin: '10px auto 6px',
            }}
          />
          <div
            style={{
              flex: 1,
              minHeight: 0,
              overflowY: 'auto',
              padding: '0 14px 6px',
            }}
          >
            {children}
          </div>
        </Drawer.Content>
      </Drawer.Portal>
    </Drawer.Root>
  );
}
```

Key decisions:
- `open={true}` permanently — sheet is always visible on mobile
- `modal={false}` — map stays interactive behind the sheet (SHEET-03)
- `dismissible={false}` — cannot drag below peek to dismiss
- `snapPoints={[80, 0.45, 1]}` — peek/half/full (SHEET-01)
- `Drawer.Handle` — visible pill with built-in 44px hitarea (SHEET-02)
- No `snapToSequentialPoint` prop — velocity flick enabled by default (SHEET-05)
- vaul's built-in `cubic-bezier(0.32, 0.72, 0, 1)` at 500ms provides spring-like animation (SHEET-04)
- `fadeFromIndex={2}` — overlay only appears at full snap
- No `Drawer.Overlay` rendered — avoids any pointer-events issues at peek/half
- `onOpenChange` safety net re-snaps to peek if sheet tries to close
- Content wrapper uses `flex: 1; min-height: 0; overflowY: auto` for scrollable content at any snap height
- Container styles match existing MobileSidebar visual treatment (same background, backdrop, border radius, safe areas)

Create `src/components/BottomSheet/__tests__/` directory and `BottomSheet.test.tsx`.

The tests verify component configuration (props passed to vaul), not vaul's internal behavior. vaul is a black box — we test that we configured it correctly.

**Test setup:** Import `{ render, screen }` from `@testing-library/react`, `{ describe, it, expect, vi }` from `vitest`, and the `BottomSheet` component.

vaul renders via a Portal, so `Drawer.Content` may not appear in the default render container. Use `screen.getByTestId` with the `data-testid` attributes added in Task 1. If Portal rendering causes issues in jsdom (no real DOM portal support), mock `Drawer.Portal` to render children inline.

**Mock strategy:** Mock the entire `vaul` module to capture the props passed to `Drawer.Root`, `Drawer.Content`, and `Drawer.Handle`. This lets us assert configuration without depending on vaul's internal DOM.

```tsx
import { render, screen } from '@testing-library/react';
import { describe, it, expect, vi } from 'vitest';

// Capture props passed to Drawer.Root
let capturedRootProps: Record<string, unknown> = {};

vi.mock('vaul', () => {
  const Drawer = {
    Root: (props: Record<string, unknown>) => {
      capturedRootProps = props;
      return <div data-testid="drawer-root">{props.children as React.ReactNode}</div>;
    },
    Portal: ({ children }: { children: React.ReactNode }) => <>{children}</>,
    Content: (props: Record<string, unknown>) => (
      <div data-testid="bottom-sheet-content" {...props}>{props.children as React.ReactNode}</div>
    ),
    Handle: (props: Record<string, unknown>) => (
      <div data-testid="bottom-sheet-handle" {...props} />
    ),
    Overlay: () => null,
  };
  return { Drawer };
});

import { BottomSheet } from '../BottomSheet';

describe('BottomSheet', () => {
  beforeEach(() => {
    capturedRootProps = {};
  });

  it('renders children inside the sheet', () => {
    render(
      <BottomSheet>
        <span>Test Content</span>
      </BottomSheet>
    );
    expect(screen.getByText('Test Content')).toBeInTheDocument();
  });

  it('configures snap points with peek, half, and full values (SHEET-01)', () => {
    render(<BottomSheet><div /></BottomSheet>);
    const snapPoints = capturedRootProps.snapPoints as (number | string)[];
    expect(snapPoints).toHaveLength(3);
    expect(snapPoints[0]).toBe(80);          // peek: 80px
    expect(snapPoints[1]).toBeCloseTo(0.45); // half: ~45vh
    expect(snapPoints[2]).toBe(1);           // full: full height
  });

  it('renders Drawer.Handle for drag pill (SHEET-02)', () => {
    render(<BottomSheet><div /></BottomSheet>);
    expect(screen.getByTestId('bottom-sheet-handle')).toBeInTheDocument();
  });

  it('sets modal=false so map stays interactive (SHEET-03)', () => {
    render(<BottomSheet><div /></BottomSheet>);
    expect(capturedRootProps.modal).toBe(false);
  });

  it('does not set snapToSequentialPoint, enabling velocity flick (SHEET-05)', () => {
    render(<BottomSheet><div /></BottomSheet>);
    expect(capturedRootProps.snapToSequentialPoint).toBeUndefined();
  });

  it('sets open=true and dismissible=false for always-visible sheet', () => {
    render(<BottomSheet><div /></BottomSheet>);
    expect(capturedRootProps.open).toBe(true);
    expect(capturedRootProps.dismissible).toBe(false);
  });

  it('starts at peek snap point', () => {
    render(<BottomSheet><div /></BottomSheet>);
    expect(capturedRootProps.activeSnapPoint).toBe(80);
  });

  it('provides onOpenChange handler to prevent sheet from closing', () => {
    render(<BottomSheet><div /></BottomSheet>);
    expect(typeof capturedRootProps.onOpenChange).toBe('function');
  });
});
```

**Note on SHEET-04:** Spring animation is internal to vaul (`cubic-bezier(0.32, 0.72, 0, 1)` at 500ms). It cannot be unit-tested — it requires visual inspection during UAT. No test needed.
